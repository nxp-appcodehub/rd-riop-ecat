PROGRAM MAIN
VAR
	Delay: TON;
	LedIn AT %I* : BOOL;
	LedOut AT %Q* : BOOL;
	// mode: 3 - AFE internal measurement
	a1o_mode AT %Q* : UDINT;
	// internal reference1.25V
	a10i AT %I* : REAL;
	// power voltage 3.3V
	a11i AT %I* : REAL;
	// power voltage +15V
	a12i AT %I* : REAL;
	// power voltage -15V
	a13i AT %I* : REAL;
END_VAR



Delay(IN:=TRUE, PT:=T#1S);
IF Delay.Q THEN
	a1o_mode := 3;
	LedOut:= NOT LedIn;
	Delay(IN:=FALSE);
END_IF;
