PROGRAM MAIN
VAR
	Delay: TON;
	LedIn AT %I* : BOOL;
	LedOut AT %Q* : BOOL;
	// mode: 6 - AFE temperature input (PT100)
	a1o_mode AT %Q* : UDINT;
	// temperature #ai4, differential, PT100, +5/J17,-6/J17, see doc.
	a14i AT %I* : REAL;
END_VAR



Delay(IN:=TRUE, PT:=T#1S);
IF Delay.Q THEN
	a1o_mode := 6;
	LedOut:= NOT LedIn;
	Delay(IN:=FALSE);
END_IF;
