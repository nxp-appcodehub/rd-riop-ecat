PROGRAM MAIN
VAR
	Delay: TON;
	LedIn AT %I* : BOOL;
	LedOut AT %Q* : BOOL;
	// mode: 1 - AFE voltage measurement
	a1o_mode AT %Q* : UDINT;
	// voltage #ai1, single end, 2/J18
	a5i AT %I* : REAL;
	// voltage #ai2, single end, 3/J18
	a6i AT %I* : REAL;
END_VAR



Delay(IN:=TRUE, PT:=T#1S);
IF Delay.Q THEN
	a1o_mode := 1;
	LedOut:= NOT LedIn;
	Delay(IN:=FALSE);
END_IF;
