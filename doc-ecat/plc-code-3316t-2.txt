PROGRAM MAIN
VAR
	Delay: TON;
	LedIn AT %I* : BOOL;
	LedOut AT %Q* : BOOL;
	// mode: 2 - AFE current input
	a1o_mode AT %Q* : UDINT;
	// current #ai3, differential, 100R shunt, +2/J17,-3/J17
	a8i AT %I* : REAL;
END_VAR



Delay(IN:=TRUE, PT:=T#1S);
IF Delay.Q THEN
	a1o_mode := 2;
	LedOut:= NOT LedIn;
	Delay(IN:=FALSE);
END_IF;
